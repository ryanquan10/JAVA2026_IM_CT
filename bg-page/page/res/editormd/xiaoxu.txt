<script type="text/javascript">
	var blogeditor = editormd("blogeditor", {
        htmlDecode:true,
		width: "100%",
		height: 700,
		path: 'assets/editormd/lib/',
		codeFold: true,
		saveHTMLToTextarea: true, // 保存 HTML 到 Textarea
		searchReplace: true,
		emoji: false,
		placeholder: "话题内容等你填写...",
		taskList: true,
		atLink: false,
		emailLink: false,
		tocm: true, // Using [TOCM]
		tex: true, // 开启科学公式TeX语言支持，默认关闭
		flowChart: true, // 开启流程图支持，默认关闭
		sequenceDiagram: true, // 开启时序/序列图支持，默认关闭,
		imageUpload: true,
		imageFormats: ["jpg", "jpeg", "gif", "png", "bmp", "webp",],
		imageUploadURL: "/p/imageUpload",
		toolbarIcons: function() {
			return [
				"undo", "redo", "|",
				"bold", "del", "italic", "quote", "ucwords", "uppercase", "lowercase", "|",
				"h1", "h2", "h3", "h4", "h5", "h6", "|",
				"list-ul", "list-ol", "hr", "|",
				"link", "reference-link", "image", "code", "preformatted-text", "code-block", "table", "datetime", "html-entities", "pagebreak", "|",
				"goto-line", "watch", "preview", "fullscreen",
			]
		},
		lineNumbers: false
	});
	$(function() {
		$("#blogeditor").on("paste", "textarea", function(e) {
			processParseImg(e, blogeditor);
		});

		// 绑定beforeunload事件
		$(window).bind('beforeunload', function() {
			return '您输入的内容尚未保存，确定离开此页面吗？';
		});
		$(".submitTopicPublishBtn").on("click", function() {
			var btn = $(this);
			if(btn.attr("ing")) {
				return false;
			}
			btn.attr("ing", true);
			var form = $("#publishTopicForm");
			if(FormChecker.check(form)) {
				MsgBox.loading("话题发布中...");
				var postData = form.serialize();
				Ajax.post("#(isEdit ? '/p/update' : '/p/save')", postData, function(data) {
					var msg = "发布成功";
					var time = 1000;
					MsgBox.success(msg, time, function() {
						// 解除绑定，一般放在提交触发事件中
						$(window).unbind('beforeunload');
						btn.removeAttr("ing", true);
						window.location.href = "p/" + data.id;
					});
				}, function() {
					btn.removeAttr("ing", true);
				});
			} else {
				btn.removeAttr("ing", true);
			}
		});
	});
</script>
